% 亮色模式颜色
\newcommand{\setLightMode}{
  \setbeamercolor{background canvas}{bg=lightBg}
  \setbeamercolor{normal text}{fg=navyPrimary, bg=lightBg}
  \setbeamercolor{frametitle}{bg=navyPrimary, fg=white}
  \setbeamercolor{block title}{bg=royalBlue, fg=white}
  \setbeamercolor{block body}{bg=white, fg=navyPrimary}
  \setbeamercolor{block title alerted}{bg=alertRed, fg=white}
  \setbeamercolor{block body alerted}{bg=white, fg=navyPrimary}
  \setbeamercolor{block title example}{bg=exampleGreen, fg=white}
  \setbeamercolor{block body example}{bg=white, fg=navyPrimary}
  \setbeamercolor{item}{fg=royalBlue}
  \setbeamercolor{subitem}{fg=skyAccent}
  \setbeamercolor{alerted text}{fg=alertRed}
  \setbeamercolor{footline}{bg=navyPrimary, fg=white}
}

% 暗色模式颜色
\newcommand{\setDarkMode}{
  \setbeamercolor{background canvas}{bg=darkBg}
  \setbeamercolor{normal text}{fg=white, bg=darkBg}
  \setbeamercolor{frametitle}{bg=darkBg2, fg=skyAccent}
  \setbeamercolor{block title}{bg=royalBlue, fg=white}
  \setbeamercolor{block body}{bg=darkBg2, fg=white!90}
  \setbeamercolor{block title alerted}{bg=alertRed, fg=white}
  \setbeamercolor{block body alerted}{bg=darkBg2, fg=white!90}
  \setbeamercolor{block title example}{bg=exampleGreen, fg=white}
  \setbeamercolor{block body example}{bg=darkBg2, fg=white!90}
  \setbeamercolor{item}{fg=skyAccent}
  \setbeamercolor{subitem}{fg=skyAccent!70}
  \setbeamercolor{alerted text}{fg=skyAccent}
  \setbeamercolor{footline}{bg=darkBg2, fg=white!70}
}

% 初始化为亮色
\setLightMode

% ---------- 暗帧环境 ----------
\newenvironment{darkframes}{
  \setDarkMode
  \setbeamertemplate{frametitle}{
    \vspace{0.4em}
    {\color{skyAccent}\rule{0.06\textwidth}{0.8pt}}
    \hspace{0.3em}\textbf{\large \insertframetitle}
    \par\vspace{0.1em}
  }
}{
  \setLightMode
  \setbeamertemplate{frametitle}{
    \vspace{0.3em}
    \textbf{\large \insertframetitle}\\[-0.15em]
    {\color{skyAccent}\rule{\textwidth}{0.6pt}}
    \vspace{0.1em}
  }
}

% ---------- 帧标题模板（亮色） ----------
\setbeamertemplate{frametitle}{
  \vspace{0.3em}
  \textbf{\large \insertframetitle}\\[-0.15em]
  {\color{skyAccent}\rule{\textwidth}{0.6pt}}
  \vspace{0.1em}
}

% ---------- 页脚 ----------
\setbeamertemplate{footline}{
  \begin{beamercolorbox}[wd=\paperwidth, ht=0.5cm, dp=0.2cm]{footline}
    \begin{tikzpicture}[overlay, remember picture]
      \fill[navyPrimary] (0,0) rectangle (\paperwidth, 0.7cm);
      \fill[skyAccent] (0, 0.65cm) rectangle (\paperwidth, 0.7cm);
      \node[anchor=west, text=white!80, font=\tiny] at (0.3cm, 0.35cm)
        {\insertshortauthor \ | \ \insertshorttitle};
      \node[anchor=east, text=white!80, font=\tiny] at (\paperwidth-0.3cm, 0.35cm)
        {\insertframenumber\,/\,\inserttotalframenumber};
    \end{tikzpicture}
  \end{beamercolorbox}
}
