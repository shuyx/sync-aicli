% 封面命令（仿 IPLeiria classic 风格）
% 版式：文档类型 → 主标题 → 分隔线 → 参与方 / 版本 / 作者 / 日期 → 摘要
%
% 用法：\reporttitle{文档类型}{主标题}{参与方A}{参与方B}{版本}{作者}{日期}{摘要}
% ============================================================
\newcommand{\reporttitle}[8]{%
  \begin{titlepage}
    \thispagestyle{empty}
    \setlength{\parindent}{0pt}
    % 全页深蓝背景（TikZ overlay，不影响 geometry）
    \begin{tikzpicture}[overlay, remember picture]
      \fill[navyBlue] (current page.south west) rectangle (current page.north east);
    \end{tikzpicture}%
    \color{white}%
    %
    \vspace*{2.5cm}
    %
    % 文档类型标签（顶部小字）
    \noindent{\small\bfseries\color{white!50}\MakeUppercase{#1}}\par
    \vspace{1.2cm}%
    %
    % 主标题（IPLeiria：\makebox[\textwidth][l]{\parbox{...}}）
    \noindent
    \makebox[\textwidth][l]{%
      \parbox{\dimexpr\textwidth-1cm\relax}{%
        \setstretch{1.05}%
        \raggedright\bfseries\fontsize{28}{34}\selectfont\color{white}
        #2%
      }%
    }\par
    \vspace{0.8cm}%
    %
    % 分隔线
    \noindent{\color{white!30}\rule{\linewidth}{0.6pt}}\par
    \vspace{0.8cm}%
    %
    % 参与方
    \noindent{\normalsize\color{white!80} #3}\par
    \vspace{0.2cm}%
    \noindent{\normalsize\color{white!70} #4}\par
    \vspace{0.5cm}%
    %
    % 版本 / 作者 / 日期
    \noindent{%
      \small\color{white!55}
      \textbf{\color{white!70}版本}\enspace #5
      \enspace\textbar\enspace
      \textbf{\color{white!70}作者}\enspace #6
      \enspace\textbar\enspace
      #7%
    }\par
    \vspace{2cm}%
    %
    % 摘要区
    \noindent
    \begin{tcolorbox}[
      enhanced,
      colback=white!6!navyBlue,
      colframe=white!20,
      toprule=0.5pt, bottomrule=0.5pt, leftrule=0.5pt, rightrule=0.5pt,
      arc=2pt, left=14pt, right=14pt, top=10pt, bottom=12pt,
      width=\linewidth,
      fontupper=\small\color{white!80}]
      \noindent{\color{white!50}\small\bfseries\MakeUppercase{Abstract}}\par
      \vspace{5pt}
      #8
    \end{tcolorbox}
    \vfill
  \end{titlepage}
  \setlength{\parindent}{2em}%
}
