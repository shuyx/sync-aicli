% --- 封面页 ---
\newcommand{\CustomTitlePage}{
  \begingroup
  \setbeamertemplate{footline}{}
  \setbeamertemplate{frametitle}{}
  \begin{frame}[noframenumbering]
    \begin{tikzpicture}[remember picture, overlay]
      % 深色背景
      \fill[darkHeader] (current page.south west) rectangle (current page.north east);
      % 橙色横条（顶部）
      \fill[orangeAccent] (current page.north west)
        rectangle ([yshift=-0.4cm]current page.north east);
      % 橙色竖条（左侧）
      \fill[orangeAccent] (current page.south west)
        rectangle ([xshift=0.15cm]current page.north west);
      % 标题文字区
      \node[anchor=west, text=white, align=left,
            font=\bfseries\Large, text width=0.75\paperwidth]
        at ([xshift=0.8cm, yshift=1.0cm]current page.west)
        {\inserttitle};
      \node[anchor=west, text=orangeAccent, align=left,
            font=\normalsize, text width=0.75\paperwidth]
        at ([xshift=0.8cm, yshift=0.2cm]current page.west)
        {\insertsubtitle};
      \node[anchor=west, text=white!80, align=left,
            font=\small, text width=0.75\paperwidth]
        at ([xshift=0.8cm, yshift=-0.7cm]current page.west)
        {\insertauthor};
      \node[anchor=west, text=white!50, align=left, font=\small]
        at ([xshift=0.8cm, yshift=-1.4cm]current page.west)
        {\insertdate};
    \end{tikzpicture}
  \end{frame}
  \endgroup
}

% --- 双栏帧环境 ---
\newenvironment{SplitFrame}[1]{
  \begin{frame}{#1}
  \begin{columns}[T, totalwidth=\textwidth]
  \begin{column}{0.47\textwidth}
}{
  \end{column}
  \end{columns}
  \end{frame}
}

\newcommand{\nextcolumn}{
  \end{column}
  \begin{tikzpicture}
    \draw[orangeAccent, line width=1.5pt] (0,-3.5) -- (0,0.3);
  \end{tikzpicture}
  \begin{column}{0.47\textwidth}
}

% --- 大字标语帧 ---
\newcommand{\BigHeadingFrame}[1]{
  \begingroup
  \setbeamertemplate{footline}{}
  \begin{frame}[noframenumbering]
    \begin{tikzpicture}[remember picture, overlay]
      \fill[darkHeader] (current page.south west) rectangle (current page.north east);
      \fill[orangeAccent] (current page.north west)
        rectangle ([yshift=-0.3cm]current page.north east);
      \node[text=white, align=center, font=\bfseries\LARGE,
            text width=0.8\paperwidth]
        at (current page.center) {#1};
    \end{tikzpicture}
  \end{frame}
  \endgroup
}

% --- 全图帧 ---
\newcommand{\FullImageFrame}[4]{
  \begingroup
  \setbeamertemplate{footline}{}
  \begin{frame}[noframenumbering]
    \begin{center}
      \includegraphics[width=#2]{#1}\\[0.3em]
      {\small\color{mutedText} #3}
    \end{center}
  \end{frame}
  \endgroup
}

% --- 致谢帧 ---
\newcommand{\ThankYouFrame}[3]{
  \begingroup
  \setbeamertemplate{footline}{}
  \begin{frame}[noframenumbering]
    \begin{tikzpicture}[remember picture, overlay]
      \fill[darkHeader] (current page.south west) rectangle (current page.north east);
      \fill[orangeAccent] (current page.north west)
        rectangle ([yshift=-0.3cm]current page.north east);
      \fill[orangeAccent] (current page.south west)
        rectangle ([xshift=0.12cm]current page.north west);

      \node[text=white, align=left, font=\bfseries\Large,
            anchor=north west]
        at ([xshift=0.8cm, yshift=-1.2cm]current page.north west)
        {#1};
      \node[text=orangeAccent, align=left, font=\normalsize, anchor=west]
        at ([xshift=0.8cm, yshift=-0.3cm]current page.west) {#2};
      \node[text=white!70, align=left, font=\small, anchor=west]
        at ([xshift=0.8cm, yshift=-1.1cm]current page.west) {#3};
    \end{tikzpicture}
  \end{frame}
  \endgroup
}
