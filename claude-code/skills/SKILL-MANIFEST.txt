⚡ 最快开始方式（5 分钟）
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. 验证 Chrome:
   /Applications/Google\ Chrome.app/Contents/MacOS/Google\ Chrome --version

2. 安装依赖:
   pip install markdown

3. 创建 test.md:
   # Hello PDF
   This is a test.

4. 运行:
   bash ~/.claude/skills/md-to-pdf-quickstart.sh test.md

5. 完成! ✅
   test.pdf 已生成

🚀 使用示例
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

【示例 1】基础使用
  bash ~/.claude/skills/md-to-pdf-quickstart.sh report.md

【示例 2】指定输出
  bash ~/.claude/skills/md-to-pdf-quickstart.sh report.md output.pdf

【示例 3】批量转换
  for f in *.md; do
    bash ~/.claude/skills/md-to-pdf-quickstart.sh "$f"
  done

【示例 4】创建别名
  alias md2pdf='bash ~/.claude/skills/md-to-pdf-quickstart.sh'
  md2pdf report.md
